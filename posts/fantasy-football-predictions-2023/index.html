With the 2023 NFL Draft having concluded, most teams are prepared to head into the 2023 NFL season with exciting young players to build around for the long-term future. Do die hard fans care about their growth and development? Obviously. Do they care even more about how it impacts their fantasy position? You better believe it.

Every year, millions of people create a fantasy football league for their friends, family, and colleagues to compete in a weekly contest of football knowledge. However, the trick for having a really successful fantasy team is by drafting rookies who many believe to be sleepers in terms of their fantasy production.

In this post, I will attempt to reveal the 5 best rookie wide receivers that will produce the most fantasy points during the 2023 NFL season. I will predict the best fantasy production of rookie wide receivers based on the following: total yards per reception, average length of catch, total touchdowns, and their pre-draft grade.

library(tidyverse)

wrstats <- read_csv("https://mattwaite.github.io/sportsdatafiles/wrdraftedstatswithratingsfixed.csv")

wrselected <- wrstats%>%
  select(
    name,
    year,
    college_team,
    nfl_team,
    overall,
    total_ypr,
    average_long,
    total_touchdowns,

  ) %>% na.omit()

player_split <- initial_split(wrselected, prop = .8)

player_train <- training(player_split)
player_test <- testing(player_split)

player_recipe <- 
  recipe(FantPt ~ ., data = player_train) %>%
  update_role(name, year, college_team, nfl_team, new_role = "ID")

xg_mod <- boost_tree(
  trees = tune(), 
  learn_rate = tune(),
  tree_depth = tune(), 
  min_n = tune(),
  loss_reduction = tune(), 
  sample_size = tune(), 
  mtry = tune(), 
  ) %>% 
  set_mode("regression") %>% 
  set_engine("xgboost")

lightgbm_mod <- 
  boost_tree() %>%
  set_engine("lightgbm") %>%
  set_mode(mode = "regression")

xg_workflow <- 
  workflow() %>% 
  add_model(xg_mod) %>% 
  add_recipe(player_recipe)

lightgbm_workflow <- 
  workflow() %>% 
  add_model(lightgbm_mod) %>% 
  add_recipe(player_recipe)

xgb_grid <- grid_latin_hypercube(
  trees(),
  tree_depth(),
  min_n(),
  loss_reduction(),
  sample_size = sample_prop(),
  finalize(mtry(), player_train),
  learn_rate()
)

player_folds <- vfold_cv(player_train)

xgb_res <- tune_grid(
  xg_workflow,
  resamples = player_folds,
  grid = xgb_grid,
  control = control_grid(save_pred = TRUE)
)

best_rmse <- select_best(xgb_res, "rmse")

final_xgb <- finalize_workflow(
  xg_workflow,
  best_rmse
)

xg_fit <- 
  final_xgb %>% 
  fit(data = player_train)

lightgbm_fit <- 
  lightgbm_workflow %>% 
  fit(data = player_train)

xgpredict <- 
  xg_fit %>% 
  predict(new_data = player_train) %>%
  bind_cols(player_train) 

lightgbmpredict <- 
  lightgbm_fit %>% 
  predict(new_data = player_train) %>%
  bind_cols(player_train)

currentdraft <- cfbd_draft_picks(2023) %>% filter(position == "Wide Receiver")

stats18 <- cfbd_stats_season_player(2018, category = "receiving") %>% remove_empty(which=c("cols")) %>% mutate(year = 2018)

stats19 <- cfbd_stats_season_player(2019, category = "receiving") %>% remove_empty(which=c("cols")) %>% mutate(year = 2019)

stats20 <- cfbd_stats_season_player(2020, category = "receiving") %>% remove_empty(which=c("cols")) %>% mutate(year = 2020)

stats21 <- cfbd_stats_season_player(2021, category = "receiving") %>% remove_empty(which=c("cols")) %>% mutate(year = 2021)

stats22 <- cfbd_stats_season_player(2022, category = "receiving") %>% remove_empty(which=c("cols")) %>% mutate(year = 2022)

current_stats <- bind_rows(stats18, stats19, stats20, stats21, stats22)

currentrating <- cfbd_ratings_sp(2022) %>% mutate(draft_year = year+1)

currenttotalstats <- current_stats %>% 
  group_by(athlete_id) %>%
  summarize(
    total_receptions = sum(receiving_rec),
    total_yards = sum(receiving_yds),
    total_touchdowns = sum(receiving_td),
    average_long = mean(receiving_long),
  ) %>%
  mutate(
    total_ypr = total_yards/total_receptions,
    athlete_id = as.numeric(athlete_id)
  )

currentdrafted <- currentdraft %>% inner_join(currenttotalstats, by=c("college_athlete_id"="athlete_id")) 

currentwrdraftclass <- currentdrafted %>% inner_join(currentrating, by=c("college_team"="team", "year"="draft_year"))

currentwrselected <- currentwrdraftclass %>%
  select(
    name,
    year,
    college_team,
    nfl_team,
    overall,
    average_long,
    total_ypr,
    total_touchdowns,
    pre_draft_grade
  ) %>% na.omit()

currentxgpredict <- 
  xg_fit %>% 
  predict(new_data = currentwrselected) %>%
  bind_cols(currentwrselected) 

currentxgpredict %>% 
  top_n(5, wt=.pred) %>%
